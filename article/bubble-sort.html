冒泡排序

2020-3-21 BY zhouzh1 
/**
 * 冒泡排序，时间复杂度为O(N^2)
 */
function bubbleSort(arr) {
    for (i = arr.length - 1; i > 0; i--) {
        for (j = 0; j < i; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
}
// test
const arr = new Array(10);
for (let i = 0; i < arr.length; i++) {
    arr[i] = Math.round(Math.random() * 80 + 20)
}
console.log(arr);
bubbleSort(arr);
console.log(arr);
